@startuml
actor Библиотекарь as bookman
database Система as system
alt Книга есть в наличии
	bookman -> system : Библиотекарь открывает форму взятия книги
	system --> bookman : Форма взятия книги открыта
	bookman -> system : Библиотекарь вносит и фиксирует данные читателя и выбранные книги
	system --> bookman : Система зафиксировала данные читателя и выбранные книги
else Книги нет в наличии
end
@enduml
